4. Alert when gas > 2x rolling average

Use ethers.js for RPC calls:
```javascript
import { ethers } from 'ethers';

const AMOY_RPC = 'https://rpc-amoy.polygon.technology';

class GasMonitor {
  constructor(alertService) {
    this.alertService = alertService;
    this.threshold = 200; // gwei
    this.history = [];
  }

  async check() {
    const provider = new ethers.JsonRpcProvider(AMOY_RPC);
    const feeData = await provider.getFeeData();
    const gasPrice = Number(ethers.formatUnits(feeData.gasPrice, 'gwei'));
    
    this.history.push({ time: Date.now(), gas: gasPrice });
    // Keep last 24h
    this.history = this.history.filter(h => Date.now() - h.time < 86400000);
    
    if (gasPrice > this.threshold) {
      await this.alertService.send({
        type: 'GAS_HIGH',
        message: `Gas price ${gasPrice} gwei exceeds threshold ${this.threshold}`,
        severity: 'warning'
      });
    }
    
    return { current: gasPrice, avg: this.avg24h };
  }

  get avg24h() {
    if (this.history.length === 0) return 0;
    return this.history.reduce((sum, h) => sum + h.gas, 0) / this.history.length;
  }
}
```

class EventWatcher {
  constructor(provider, contracts, wsServer) {
    this.provider = provider;
    this.contracts = contracts;
    this.wsServer = wsServer;
    this.recentEvents = [];
  }

  async start() {
    // Watch StakingManager events
    const staking = new ethers.Contract(stakingAddr, stakingAbi, this.provider);
    staking.on('Stake', (user, skillId, amount, event) => {
      this.recordEvent('Stake', { user, skillId: skillId.toString(), amount: amount.toString() });
    });
    
    // Watch SkillRegistry events
    const skillRegistry = new ethers.Contract(skillAddr, skillAbi, this.provider);
    skillRegistry.on('SkillRegistered', (skillId, owner, name, event) => {
      this.recordEvent('SkillRegistered', { skillId: skillId.toString(), owner, name });
    });
    
    // ... add other contracts similarly
  }

  recordEvent(name, data) {
    const event = {
      name,
      data,
      timestamp: Date.now(),
      blockNumber: event.log.blockNumber
    };
    
    this.recentEvents.push(event);
    if (this.recentEvents.length > 1000) {
      this.recentEvents.shift();
    }
    
    // Emit to WebSocket
    if (this.wsServer) {
      this.wsServer.clients.forEach(client => {
        client.send(JSON.stringify(event));
      });
    }
  }
}

2. Alert types:

   - SLASH_EVENT: Someone slashed a stake
   - SKILL_SLASHED: Skill verification failed
   - PROPOSAL_CREATED: New governance proposal
   - HEALTH_REPORT: Health report submitted
   - ERROR: Script errors

3. Rate limiting: Max 1 alert per type per 5 minutes

```javascript
class AlertService {
  constructor(config) {
    this.email = config.email; // { enabled, smtp, to }
    this.telegram = config.telegram; // { enabled, botToken, chatId }
    this.webhook = config.webhook; // { enabled, url, headers }
    this.lastAlert = {}; // type -> timestamp
    this.rateLimitMs = 300000; // 5 minutes
  }

  async send(alert) {
    // Rate limit check
    const now = Date.now();
    if (this.lastAlert[alert.type] && now - this.lastAlert[alert.type] < this.rateLimitMs) {
      console.log(`Alert ${alert.type} rate limited, skipping`);
      return;
    }
    this.lastAlert[alert.type] = now;

    // Send to all channels
    const promises = [];
    
    if (this.console.enabled) {
      promises.push(this.sendConsole(alert));
    }
    if (this.email?.enabled) {
      promises.push(this.sendEmail(alert));
    }
    if (this.telegram?.enabled) {
      promises.push(this.sendTelegram(alert));
    }
    if (this.webhook?.enabled) {
      promises.push(this.sendWebhook(alert));
    }
    
    await Promise.allSettled(promises);
  }

  async sendTelegram(alert) {
    const message = `*[${alert.type}]* ${alert.message}\nSeverity: ${alert.severity || 'info'}`;
    await fetch(`https://api.telegram.org/bot${this.telegram.botToken}/sendMessage`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        chat_id: this.telegram.chatId,
        text: message,
        parse_mode: 'Markdown'
      })
    });
  }

  async sendWebhook(alert) {
    await fetch(this.webhook.url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', ...this.webhook.headers },
      body: JSON.stringify(alert)
    });
  }
}
```

function OpsDashboard() {
  const [gasData, setGasData] = useState({ current: 0, avg: 0 });
  const [events, setEvents] = useState([]);
  
  useEffect(() => {
    // Poll gas data every 30s
    const gasInterval = setInterval(async () => {
      const data = await fetch('/api/monitoring/gas').then(r => r.json());
      setGasData(data);
    }, 30000);
    
    // WebSocket for events
    const ws = new WebSocket('ws://localhost:8080/events');
    ws.onmessage = (e) => {
      const event = JSON.parse(e.data);
      setEvents(prev => [event, ...prev].slice(0, 50));
    };
    
    return () => {
      clearInterval(gasInterval);
      ws.close();
    };
  }, []);
  
  return (
    <div className="ops-dashboard">
      <div className="gas-panel">
        <GasGauge value={gasData.current} avg={gasData.avg} />
        <span>24h Avg: {gasData.avg.toFixed(1)} gwei</span>
      </div>
      <div className="events-panel">
        <h3>Recent Events</h3>
        <EventFeed events={events} />
      </div>
      <div className="health-panel">
        <HealthStatus contracts={['StakingManager', 'SkillRegistry', 'Attribution']} />
      </div>
    </div>
  );
}

import express from 'express';
import { WebSocketServer } from 'ws';
import GasMonitor from './gasMonitor.js';
import EventWatcher from './eventWatcher.js';
import AlertService from './alertService.js';

const app = express();
const PORT = process.env.PORT || 3001;

// WebSocket server
const wss = new WebSocketServer({ port: 8080 });
const wsClients = new Set();

wss.on('connection', (ws) => {
  wsClients.add(ws);
  ws.on('close', () => wsClients.delete(ws));
});

// Emit events to all clients
function broadcastEvent(event) {
  wsClients.forEach(client => {
    if (client.readyState === 1) { // OPEN
      client.send(JSON.stringify(event));
    }
  });
}

// Initialize services
const alertService = new AlertService({
  telegram: {
    enabled: !!process.env.TELEGRAM_BOT_TOKEN,
    botToken: process.env.TELEGRAM_BOT_TOKEN,
    chatId: process.env.TELEGRAM_CHAT_ID
  },
  webhook: {
    enabled: !!process.env.WEBHOOK_URL,
    url: process.env.WEBHOOK_URL
  },
  console: { enabled: true }
});

const gasMonitor = new GasMonitor(alertService);
const eventWatcher = new EventWatcher(provider, contracts, { clients: wsClients });

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({ status: 'ok', timestamp: Date.now() });
});

// Gas data endpoint
app.get('/api/monitoring/gas', (req, res) => {
  res.json(gasMonitor.getLatest());
});

// Events endpoint
app.get('/api/monitoring/events', (req, res) => {
  res.json(eventWatcher.getRecentEvents());
});

app.listen(PORT, () => {
  console.log(`Monitoring server running on port ${PORT}`);
  gasMonitor.start();
  eventWatcher.start();
});
