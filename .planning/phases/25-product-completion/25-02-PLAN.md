    ```javascript
    const https = require('https');

    class GasMonitor {
      constructor() {
        this.alertThreshold = {
          fast: 100,      // Gwei
          standard: 50,   // Gwei
          slow: 20        // Gwei
        };
        this.lastPrices = [];
      }

      async fetchGasPrice() {
        // Use Polygon RPC or API
        const response = await fetch('https://gasstation-mainnet.matic.network/v2');
        const data = await response.json();
        return {
          fast: data.fast.maxPrice,
          standard: data.standard.maxPrice,
          slow: data.slow.maxPrice,
          timestamp: Date.now()
        };
      }

      async checkGasPrice() {
        const price = await this.fetchGasPrice();
        this.lastPrices.push(price);

        // Keep last 100 readings
        if (this.lastPrices.length > 100) {
          this.lastPrices.shift();
        }

        // Alert if gas is high
        if (price.fast > this.alertThreshold.fast) {
          await this.alert('HIGH_GAS', `Fast gas at ${price.fast} Gwei`);
        }

        return price;
      }

      getAverageGas(samples = 10) {
        const recent = this.lastPrices.slice(-samples);
        if (recent.length === 0) return null;

        return {
          fast: recent.reduce((a, b) => a + b.fast, 0) / recent.length,
          standard: recent.reduce((a, b) => a + b.standard, 0) / recent.length,
          slow: recent.reduce((a, b) => a + b.slow, 0) / recent.length
        };
      }
    }

    module.exports = { GasMonitor };
    ```

    ```javascript
        };
      }

      createHandlers() {
        return {
          Staked: async (event) => {
            console.log(`Stake: ${event.args.user} staked ${event.args.amount} on skill ${event.args.skillId}`);
            // Update database, send notifications, etc.
          },
          Slash: async (event) => {
            console.log(`Slash: ${event.args.user} slashed ${event.args.amount}`);
            await this.alert('SLASH', `User ${event.args.user} slashed ${event.args.amount}`);
          },
          AntiSlash: async (event) => {
            console.log(`AntiSlash: ${event.args.user} penalized ${event.args.penalty}`);
            await this.alert('ANTI_SLASH', `User ${event.args.user} penalized ${event.args.penalty}`);
          },
          // ... other handlers
        };
      }

      async start() {
        for (const [name, filter] of Object.entries(this.filters)) {
          this.contracts.stakingManager.on(filter, (...args) => {
            this.handlers[name](args[args.length - 1]); // Last arg is event
          });
        }
        console.log('Event watcher started');
      }

      async stop() {
        this.contracts.stakingManager.removeAllListeners();
        console.log('Event watcher stopped');
      }
    }

    module.exports = { EventWatcher };
    ```

    ```javascript
    class AlertService {
      constructor(config) {
        this.channels = {
          email: config.email,
          telegram: config.telegram,
          slack: config.slack
        };
        this.alertHistory = [];
      }

      async send(type, message, severity = 'INFO') {
        const alert = {
          id: Date.now(),
          type,
          message,
          severity,
          timestamp: new Date().toISOString()
        };

        this.alertHistory.push(alert);
        console.log(`[${severity}] ${type}: ${message}`);

        // Send to configured channels based on severity
        if (severity === 'CRITICAL') {
          await this.sendAllChannels(alert);
        } else if (severity === 'HIGH') {
          await this.sendTelegram(alert);
          await this.sendEmail(alert);
        } else {
          // Log only for INFO/LOW
          this.logAlert(alert);
        }

        return alert;
      }

      async sendTelegram(alert) {
        if (!this.channels.telegram) return;

        const botToken = this.channels.telegram.botToken;
        const chatId = this.channels.telegram.chatId;
        const url = `https://api.telegram.org/bot${botToken}/sendMessage`;

        await fetch(url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            chat_id: chatId,
            text: `[${alert.severity}] ${alert.type}\n${alert.message}\n${alert.timestamp}`
          })
        });
      }

      async sendEmail(alert) {
        if (!this.channels.email) return;
        // Implement email sending via SMTP or service
      }

      async sendSlack(alert) {
        if (!this.channels.slack) return;
        // Implement Slack webhook
      }

      async sendAllChannels(alert) {
        await Promise.all([
          this.sendTelegram(alert),
          this.sendEmail(alert),
          this.sendSlack(alert)
        ]);
      }

      getAlertHistory(hours = 24) {
        const cutoff = Date.now() - hours * 60 * 60 * 1000;
        return this.alertHistory.filter(a => a.timestamp > cutoff);
      }
    }

    module.exports = { AlertService };
    ```

    ```javascript
    const express = require('express');
    const { GasMonitor } = require('./services/gasMonitor');
    const { EventWatcher } = require('./services/eventWatcher');
    const { AlertService } = require('./services/alertService');
    const { ethers } = require('ethers');

    const app = express();
    app.use(express.json());

    // Initialize services
    const provider = new ethers.JsonRpcProvider(process.env.POLYGON_RPC);
    const stakingManager = new ethers.Contract(STAKING_MANAGER_ADDRESS, STAKING_ABI, provider);
    const skillRegistry = new ethers.Contract(SKILL_REGISTRY_ADDRESS, SKILL_REGISTRY_ABI, provider);
    const attribution = new ethers.Contract(ATTRIBUTION_ADDRESS, ATTRIBUTION_ABI, provider);

    const contracts = { stakingManager, skillRegistry, attribution };
    const gasMonitor = new GasMonitor();
    const eventWatcher = new EventWatcher(contracts, provider);
    const alertService = new AlertService({
      telegram: {
        botToken: process.env.TELEGRAM_BOT_TOKEN,
        chatId: process.env.TELEGRAM_CHAT_ID
      }
    });

    // Health check endpoint
    app.get('/health', (req, res) => {
      res.json({ status: 'ok', uptime: process.uptime() });
    });

    // Metrics endpoint
    app.get('/metrics', async (req, res) => {
      const avgGas = gasMonitor.getAverageGas();
      const alerts = alertService.getAlertHistory(24);

      res.json({
        gasPrice: avgGas,
        alertCount24h: alerts.length,
        alerts
      });
    });

    // Alert history endpoint
    app.get('/alerts', (req, res) => {
      const hours = parseInt(req.query.hours) || 24;
      res.json(alertService.getAlertHistory(hours));
    });

    // Start server
    const PORT = process.env.PORT || 3001;
    app.listen(PORT, async () => {
      console.log(`Monitoring server on port ${PORT}`);

      // Start monitoring
      gasMonitor.checkGasPrice();
      setInterval(() => gasMonitor.checkGasPrice(), 60000); // Every minute

      // Start event watching
      await eventWatcher.start();
    });

    // Graceful shutdown
    process.on('SIGTERM', async () => {
      console.log('Shutting down...');
      await eventWatcher.stop();
      process.exit(0);
    });
    ```
