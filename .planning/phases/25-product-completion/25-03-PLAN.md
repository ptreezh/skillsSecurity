```solidity
     */

        address[] memory proposers,
        address[] memory executors,
        address admin
    ) TimelockController(minDelay, proposers, executors, admin) {}
}
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/governance/Governor.sol";
import "@openzeppelin/contracts/governance/compatibility/GovernorCompatibilityBravo.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorVotes.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorVotesQuorumFraction.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorTimelockControl.sol";

/**
 * @title AgentGovernor
 * @dev Governance contract for AgentSkills with timelock integration.
 *      Uses reputation-based voting (no token required).
 */
contract AgentGovernor is 
    Governor,
    GovernorCompatibilityBravo,
    GovernorVotes,
    GovernorVotesQuorumFraction,
    GovernorTimelockControl
{
    // Governance parameters
    uint256 public constant VOTING_DELAY = 1;           // 1 block
    uint256 public constant VOTING_PERIOD = 45818;       // ~1 week (assuming 1 block every 1.5s)
    uint256 public constant PROPOSAL_THRESHOLD = 100;   // minimum reputation to propose
    uint256 public constant QUORUM_PERCENTAGE = 4;       // 4% of total votes

    /**
     * @param _token Governance token (use address(0) for reputation-based)
     * @param _timelock TimelockController address
     */
    constructor(
        IVotes _token,
        TimelockController _timelock
    )
        Governor("AgentSkills Governor")
        GovernorVotes(_token)
        GovernorVotesQuorumFraction(QUORUM_PERCENTAGE)
        GovernorTimelockControl(_timelock)
    {}

    // Override required functions
    function votingDelay() public view override returns (uint256) {
        return VOTING_DELAY;
    }

    function votingPeriod() public view override returns (uint256) {
        return VOTING_PERIOD;
    }

    function proposalThreshold() public view override returns (uint256) {
        return PROPOSAL_THRESHOLD;
    }

    // Override to use timelock
    function timelock() public view override(Governor, GovernorTimelockControl) returns (address) {
        return GovernorTimelockControl.timelock();
    }

    // Cancel a proposal (requires proposer or guardian)
    function cancelProposal(uint256 proposalId) external override {
        // Implementation
    }
}
```

```javascript
const { ethers } = require('hardhat');

const DELAY = 172800; // 48 hours in seconds
const VOTING_DELAY = 1; // 1 block
const VOTING_PERIOD = 45818; // ~1 week
const PROPOSAL_THRESHOLD = ethers.parseEther('100'); // 100 reputation
const QUORUM_PERCENTAGE = 4;

async function deployTimelock() {
  const [deployer, ...others] = await ethers.getSigners();
  
  console.log('Deploying AgentTimelock...');
  
  // Timelock configuration
  const proposers = [deployer.address]; // Will update to AgentGovernor after deployment
  const executors = [ethers.ZeroAddress]; // Any address can execute (or specific multisig)
  const admin = deployer.address; // Admin for role management
  
  const AgentTimelock = await ethers.getContractFactory('AgentTimelock');
  const timelock = await AgentTimelock.deploy(
    DELAY,
    proposers,
    executors,
    admin
  );
  await timelock.waitForDeployment();
  console.log(`AgentTimelock deployed to: ${timelock.target}`);
  
  console.log('Deploying ReputationVotes...');
  const ReputationVotes = await ethers.getContractFactory('ReputationVotes');
  const reputationVotes = await ReputationVotes.deploy(
    '0xYourStakingManagerAddress' // Update after StakingManager deployment
  );
  await reputationVotes.waitForDeployment();
  console.log(`ReputationVotes deployed to: ${reputationVotes.target}`);
  
  console.log('Deploying AgentGovernor...');
  const AgentGovernor = await ethers.getContractFactory('AgentGovernor');
  const governor = await AgentGovernor.deploy(
    reputationVotes.target, // governance token (ReputationVotes)
    timelock.target         // timelock controller
  );
  await governor.waitForDeployment();
  console.log(`AgentGovernor deployed to: ${governor.target}`);
  
  // Update timelock proposers to include governor
  console.log('Configuring timelock proposers...');
  const proposerRole = await timelock.PROPOSER_ROLE();
  await timelock.grantRole(proposerRole, governor.target);
  
  // Revoke deployer as proposer (optional - security)
  // await timelock.revokeRole(proposerRole, deployer.address);
  
  // Update ReputationVotes with StakingManager address
  console.log('Updating ReputationVotes with StakingManager...');
  const stakingManagerAddress = '0xYourStakingManagerAddress'; // From deployments
  await reputationVotes.updateStakingManager(stakingManagerAddress);
  
  console.log('\n=== Deployment Complete ===');
  console.log(`AgentTimelock: ${timelock.target}`);
  console.log(`AgentGovernor: ${governor.target}`);
  console.log(`ReputationVotes: ${reputationVotes.target}`);
  console.log(`Timelock Delay: 48 hours`);
  
  return { timelock, governor, reputationVotes };
}

module.exports = { deployTimelock };
```
