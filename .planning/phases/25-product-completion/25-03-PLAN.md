     */
    contract AgentTimelock is TimelockController {
        // Timelock delay: 48 hours
        uint256 private constant MIN_DELAY = 48 hours;
        uint256 private constant MAX_DELAY = 30 days;

        // Proposer and executor roles will be granted to AgentGovernor
        bytes32 public constant PROPOSER_ROLE = keccak256("PROPOSER_ROLE");
        bytes32 public constant EXECUTOR_ROLE = keccak256("EXECUTOR_ROLE");
        bytes32 public constant CANCELLER_ROLE = keccak256("CANCELLER_ROLE");

        constructor(
            uint256 minDelay,
            address[] memory proposers,
            address[] memory executors
        )
            TimelockController(minDelay, proposers, executors, msg.sender)
        {
            // Grant canceller role to multisig for emergency cancellations
        }

        function updateDelay(uint256 newDelay) external onlyRole(TIMELOCK_ADMIN_ROLE) {
            require(newDelay >= MIN_DELAY, "Delay too short");
            require(newDelay <= MAX_DELAY, "Delay too long");
            // Must go through timelock itself for security
        }
    }

    ```solidity
    // SPDX-License-Identifier: MIT
    pragma solidity ^0.8.20;

    import "@openzeppelin/contracts/governance/utils/Votes.sol";
    import "@openzeppelin/contracts/governance/utils/IVotes.sol";
    import "./StakingManager.sol";

    /**
     * @title ReputationVotes
     * @notice Voting power based on reputation in StakingManager
     */
    contract ReputationVotes is IVotes {
        StakingManager public immutable stakingManager;

        constructor(address _stakingManager) {
            stakingManager = StakingManager(_stakingManager);
        }

        /**
         * @dev Voting power is based on effective reputation
         */
        function getVotes(address account) public view override returns (uint256) {
            int256 effectiveRep = stakingManager.getUserReputation(account);
            return effectiveRep > 0 ? uint256(effectiveRep) : 0;
        }

        /**
         * @dev Returns the past votes at a specific checkpoint
         * @dev For simplicity, we use current reputation as past votes
         */
        function getPastVotes(address account, uint256 blockNumber) public view override returns (uint256) {
            return getVotes(account);
        }

        /**
         * @dev Returns the total supply of voting power
         */
        function getPastTotalSupply(uint256 blockNumber) public view override returns (uint256) {
            return getVotes(msg.sender); // Approximation
        }

        // Optional: Track checkpoints for historical voting power
        mapping(address => mapping(uint256 => Checkpoint)) private _checkpoints;
        mapping(address => uint256) private _numCheckpoints;

        struct Checkpoint {
            uint32 fromBlock;
            uint224 votes;
        }
    }
    ```

    Voting power calculation:

    - Locked reputation does NOT count (only effective)

    const hre = require("hardhat");

    async function main() {
      const [deployer] = await hre.ethers.getSigners();

      console.log("Deploying AgentSkills Governance...");

      // 1. Deploy Timelock with 48-hour delay
      const MIN_DELAY = 48 * 60 * 60; // 48 hours in seconds

      // Initially grant deployer as proposer/executor, will transfer to governor
      const timelock = await hre.ethers.deployContract("AgentTimelock", [
        MIN_DELAY,
        [deployer.address],  // proposers
        [deployer.address],    // executors
      ]);
      await timelock.waitForDeployment();
      console.log("AgentTimelock deployed to:", await timelock.getAddress());

      // 2. Deploy ReputationVotes
      const reputationVotes = await hre.ethers.deployContract("ReputationVotes", [
        process.env.STAKING_MANAGER_ADDRESS,
      ]);
      await reputationVotes.waitForDeployment();
      console.log("ReputationVotes deployed to:", await reputationVotes.getAddress());

      // 3. Deploy Governor
      const governor = await hre.ethers.deployContract("AgentGovernor", [
        await reputationVotes.getAddress(),
        await timelock.getAddress(),
      ]);
      await governor.waitForDeployment();
      console.log("AgentGovernor deployed to:", await governor.getAddress());

      // 4. Configure Timelock - grant proposer role to Governor
      const PROPOSER_ROLE = await timelock.PROPOSER_ROLE();
      const CANCELLER_ROLE = await timelock.CANCELLER_ROLE();

      await timelock.grantRole(PROPOSER_ROLE, await governor.getAddress());
      await timelock.grantRole(CANCELLER_ROLE, await governor.getAddress());

      // 5. Revoke deployer proposer role (optional - governance takes over)
      await timelock.revokeRole(PROPOSER_ROLE, deployer.address);

      console.log("\nGovernance deployment complete!");
      console.log({
        timelock: await timelock.getAddress(),
        reputationVotes: await reputationVotes.getAddress(),
        governor: await governor.getAddress(),
      });

      // 6. Save deployment addresses
      const governance = {
        network: hre.network.name,
        timelock: await timelock.getAddress(),
        reputationVotes: await reputationVotes.getAddress(),
        governor: await governor.getAddress(),
        parameters: {
          votingDelay: "1 day",
          votingPeriod: "7 days",
          proposalThreshold: "100 reputation",
          quorumPercentage: "4%",
          timelockDelay: "48 hours",
        },
      };

      const fs = require("fs");
      fs.writeFileSync(
        "deployments/governance.json",
        JSON.stringify(governance, null, 2)
      );
    }

    main().catch((error) => {
      console.error(error);
      process.exit(1);
    });
