/**
 * RevenueCollector
 * Collects protocol fees automatically
 *
 * Fee Structure:
 * - Skill usage: 0.5% of call value
 * - Verification: 0.1% of stake amount
 * - Slash penalty: 5% of slashed amount
 */

class RevenueCollector {
  constructor(config) {
    this.stakingManager = config.stakingManager
    this.treasury = config.treasury
    this.feeRates = {
      usage: 0.005,      // 0.5%
      verification: 0.001,  // 0.1%
      slash: 0.05       // 5%
    }
    this.revenue = { usage: 0, verification: 0, slash: 0 }
  }

  // Event listeners for fee collection
  async onSkillUsed(skillId, value) {
    const fee = value * this.feeRates.usage
    this.revenue.usage += fee
    await this.sweepToTreasury()
  }

  async onVerification(skillId, stakeAmount) {
    const fee = stakeAmount * this.feeRates.verification
    this.revenue.verification += fee
    await this.sweepToTreasury()
  }

  async onSlash(slashedAmount) {
    const penalty = slashedAmount * this.feeRates.slash
    this.revenue.slash += penalty
    await this.sweepToTreasury()
  }

  async sweepToTreasury() {
    const total = this.revenue.usage + this.revenue.verification + this.revenue.slash
    if (total > 0) {
      await this.treasury.deposit(total)
      this.revenue = { usage: 0, verification: 0, slash: 0 }
    }
  }

  getRevenue() {
    return { ...this.revenue }
  }
}

module.exports = { RevenueCollector }

 * StakingRewards
 * Distributes staking rewards based on lock duration
 *
 * APY Tiers:
 * - Skill creators: 8% (1 year lock)
 * - Validators: 12% (6 months lock)
 * - Liquidity providers: 5% (flexible)
 */

class StakingRewards {
  constructor(config) {
    this.stakingManager = config.stakingManager
    this.rewardPool = config.rewardPool
    this.apyTiers = {
      skillCreator: 0.08,  // 8%
      validator: 0.12,     // 12%
      liquidity: 0.05      // 5%
    }
    this.compoundThreshold = 90 * 24 * 60 * 60 * 1000 // 90 days
  }

  calculateReward(address) {
    const stake = this.stakingManager.getStake(address)
    const tier = this.getTier(address)
    const apy = this.apyTiers[tier]

    // Calculate daily reward
    const dailyRate = apy / 365
    const reward = stake.amount * dailyRate

    return {
      daily: reward,
      weekly: reward * 7,
      monthly: reward * 30,
      yearly: stake.amount * apy
    }
  }

    const reward = this.calculateReward(address)

    // Compound if lock > 90 days

      return this.compoundReward(address, reward.daily)
