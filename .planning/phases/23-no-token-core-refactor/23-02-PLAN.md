Contract structure after refactor:
```solidity
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";
import "./StakingManager.sol";

struct Skill {
    address owner;
    string name;
    string description;
    string trigger;
    string metadataIPFS;
    RiskLevel riskLevel;
    uint256 stakeAmount;      // Kept for display, no actual transfer
    bool verified;
    uint256 createdAt;
    uint256 updatedAt;
    string version;
    bytes32 fingerprint;
}

enum RiskLevel { LOW, MEDIUM, HIGH, CRITICAL }

contract SkillRegistry is Ownable {
    mapping(uint256 => Skill) public skills;
    mapping(address => uint256[]) public ownerSkills;
    mapping(uint256 => bool) public verifiedSkills;
    
    // REMOVED: ASKToken public immutable token;
    StakingManager public stakingManager;
    
    // Constructor - NO token parameter
    constructor(address _stakingManager) Ownable() {
        // No token to initialize
        stakingManager = StakingManager(_stakingManager);
    }
    
    // registerSkill() - NO token.transferFrom() call
    function registerSkill(...) external returns (uint256);
    
    // verifySkill() - uses stakingManager.getUserReputation()
    function verifySkill(uint256 _skillId, bool _pass) external;
    
    // slashSkill() - just reduces stakeAmount, no token transfer
    function slashSkill(uint256 _skillId, uint256 _amount) external onlyOwner;
}
```

- CRITICAL: 5000+ effective reputation

Verification reputation requirements:
- LOW verifier: 100+ effective reputation
- MEDIUM verifier: 500+ effective reputation
- HIGH verifier: 1000+ effective reputation
- CRITICAL verifier: 2000+ effective reputation
