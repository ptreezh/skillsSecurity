/**
 * Test Fixtures for AgentSkills Contracts
 * 
 * Deployment Order (per D-02):
 * 1. ASKToken — no dependencies
 * 2. StakingManager — requires ASKToken address
 * 3. SkillRegistry — requires ASKToken + StakingManager addresses
 * 4. Attribution — requires setStakingManager() call after deployment (per D-03)
 */

const { loadFixture } = require("@nomicfoundation/hardhat-network-helpers");

/**
 * Deploy all contracts in correct dependency order.
 * Uses loadFixture for snapshot isolation (handles beforeEach automatically).
 * 
 * @returns {Object} { token, staking, registry, attribution }
 */
async function deployContracts() {
  const [owner, user1, user2, ...accounts] = await ethers.getSigners();

  // 1. Deploy ASKToken
  const Token = await ethers.getContractFactory("ASKToken");
  const token = await Token.deploy();
  await token.deployed();

  // 2. Deploy StakingManager (requires token address)
  const Staking = await ethers.getContractFactory("StakingManager");
  const staking = await Staking.deploy(token.address);
  await staking.deployed();

  // 3. Deploy SkillRegistry (requires token + stakingManager addresses)
  const Registry = await ethers.getContractFactory("SkillRegistry");
  const registry = await Registry.deploy(token.address, staking.address);
  await registry.deployed();

  // 4. Deploy Attribution (no constructor args, but needs setStakingManager)
  const Attribution = await ethers.getContractFactory("Attribution");
  const attribution = await Attribution.deploy();
  await attribution.deployed();
  
  // CRITICAL: Wire Attribution to StakingManager (per D-03 and PITFALL #1)
  await attribution.setStakingManager(staking.address);

  return { token, staking, registry, attribution, owner, user1, user2, accounts };
}

module.exports = {
  deployContracts,
  // Also export contract names for direct use
  ASKToken: "ASKToken",
  StakingManager: "StakingManager",
  SkillRegistry: "SkillRegistry",
  Attribution: "Attribution"
};
```

Key points:
1. Use `loadFixture` pattern for snapshot isolation
2. Correct deployment order: token → staking → registry → attribution
3. Attribution.setStakingManager() MUST be called after deployment
4. Export deployContracts() for all test files to use
5. Include signers in return for convenience in tests

const { expect } = require("chai");
const { deployContracts } = require("./fixtures");

describe("Fixture Smoke Test", function() {
  // Use loadFixture for snapshot isolation
  const fixture = deployContracts;

  it("should deploy all contracts successfully", async function() {
    const { token, staking, registry, attribution } = await loadFixture(fixture);
    
    // Verify all addresses are non-zero
    expect(token.address).to.not.equal(ethers.constants.AddressZero);
    expect(staking.address).to.not.equal(ethers.constants.AddressZero);
    expect(registry.address).to.not.equal(ethers.constants.AddressZero);
    expect(attribution.address).to.not.equal(ethers.constants.AddressZero);
  });

  it("should wire contracts correctly", async function() {
    const { token, staking, registry, attribution } = await loadFixture(fixture);
    
    // Verify SkillRegistry has correct token and stakingManager
    expect(await registry.token()).to.equal(token.address);
    expect(await registry.stakingManager()).to.equal(staking.address);
    
    // Verify Attribution has stakingManager set (per D-03 and PITFALL #1)
    expect(await attribution.stakingManager()).to.equal(staking.address);
  });

  it("should have correct contract dependencies", async function() {
    const { token, staking, registry } = await loadFixture(fixture);
    
    // StakingManager should reference the token
    expect(await staking.token()).to.equal(token.address);
  });
});
```

This smoke test ensures:
1. All contracts deploy without reverting
2. Wiring is correct (SkillRegistry has token + stakingManager)
3. Attribution.setStakingManager() was called (attribution.stakingManager() returns correct address)
4. Snapshot isolation works (tests can run in any order)
