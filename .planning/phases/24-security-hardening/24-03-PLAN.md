OpenZeppelin Pausable pattern:

```solidity
// contracts/Pausable.sol - Custom implementation
abstract contract Pausable {
    bool public paused = false;
    
    event Paused(address account);
    event Unpaused(address account);
    
    modifier whenNotPaused() {
        require(!paused, "Pausable: paused");
        _;
    }
    
    modifier whenPaused() {
        require(paused, "Pausable: not paused");
        _;
    }
    
    function _pause() internal virtual {
        paused = true;
        emit Paused(msg.sender);
    }
    
    function _unpause() internal virtual {
        paused = false;
        emit Unpaused(msg.sender);
    }
}
```

Create a custom Pausable contract that integrates with GovernanceMultisig for pause control.

The contract should:
1. Inherit from nothing (standalone)
2. Store paused state
3. Have whenNotPaused and whenPaused modifiers
4. Have pause() and unpause() functions controlled by governance
5. Emit events for audit trail

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "./GovernanceMultisig.sol";

abstract contract Pausable {
    GovernanceMultisig public governance;
    bool public paused = false;
    
    event Paused(address indexed by, bytes32 indexed actionId);
    event Unpaused(address indexed by, bytes32 indexed actionId);
    
    modifier whenNotPaused() {
        require(!paused, "Pausable: paused");
        _;
    }
    
    modifier whenPaused() {
        require(paused, "Pausable: not paused");
        _;
    }
    
    function setGovernance(address _governance) internal {
        require(address(governance) == address(0), "Governance already set");
        require(_governance != address(0), "Invalid address");
        governance = GovernanceMultisig(_governance);
    }
    
    function pause(bytes32 _actionId) external {
        require(address(governance) != address(0), "Governance not set");
        require(governance.isAuthorized(msg.sender, _actionId), "Not authorized");
        require(!paused, "Already paused");
        
        paused = true;
        emit Paused(msg.sender, _actionId);
    }
    
    function unpause(bytes32 _actionId) external {
        require(address(governance) != address(0), "Governance not set");
        require(governance.isAuthorized(msg.sender, _actionId), "Not authorized");
        require(paused, "Not paused");
        
        paused = false;
        emit Unpaused(msg.sender, _actionId);
    }
}
```
