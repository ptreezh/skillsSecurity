  <action>
Create a new file `contracts/GovernanceMultisig.sol` implementing multi-signature governance.

The contract should implement:

1. **Initialization (constructor)**:
   - Accept array of signer addresses (minimum 3, maximum 10)
   - Store signers and set isSigner mapping
   - Emit events for each signer

2. **Action submission**:
   - submitAction(address _target, bytes _data, string _description) → returns actionId
   - Generates unique actionId using keccak256(target, data, block.timestamp, msg.sender)
   - Stores action with 24-hour execute delay
   - Emits ActionSubmitted event

3. **Action confirmation**:
   - confirmAction(bytes32 _actionId)
   - Only signers can confirm
   - Each signer can only confirm once
   - When confirmations >= REQUIRED_CONFIRMATIONS (3), action becomes executable
   - Emits ActionConfirmed event

4. **Action execution**:
   - executeAction(bytes32 _actionId)
   - Only executable after delay (24 hours)
   - Only if enough confirmations
   - Calls target with data
   - Marks as executed

5. **Authorization check**:
   - isAuthorized(address _caller, bytes32 _actionId) → returns bool
   - Returns true if: caller is signer AND action has required confirmations AND not executed

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/access/Ownable.sol";

contract GovernanceMultisig {
    uint256 public constant REQUIRED_CONFIRMATIONS = 3;
    uint256 public constant ACTION_DELAY = 24 hours;
    uint256 public constant MIN_SIGNERS = 3;
    uint256 public constant MAX_SIGNERS = 10;

    address[] public signers;
    mapping(address => bool) public isSigner;

    struct Action {
        address target;
        bytes data;
        uint256 value;
        string description;
        uint256 confirmations;
        uint256 executeAfter;
        bool executed;
    }

    mapping(bytes32 => Action) public pendingActions;
    mapping(bytes32 => mapping(address => bool)) public confirmations;
    bytes32[] public actionIds;

    event ActionSubmitted(bytes32 indexed actionId, address indexed target, string description);
    event ActionConfirmed(bytes32 indexed actionId, address indexed signer);
    event ActionExecuted(bytes32 indexed actionId);
    event ActionCancelled(bytes32 indexed actionId);
    event SignerAdded(address indexed signer);

    constructor(address[] memory _signers) {
        require(_signers.length >= MIN_SIGNERS, "Need minimum signers");
        require(_signers.length <= MAX_SIGNERS, "Too many signers");
        
        for (uint256 i = 0; i < _signers.length; i++) {
            require(_signers[i] != address(0), "Invalid signer");
            require(!isSigner[_signers[i]], "Duplicate signer");
            
            signers.push(_signers[i]);
            isSigner[_signers[i]] = true;
            emit SignerAdded(_signers[i]);
        }
    }

    function submitAction(
        address _target,
        bytes calldata _data,
        string calldata _description
    ) external returns (bytes32) {
        require(isSigner[msg.sender], "Only signers can submit");
        
        bytes32 actionId = keccak256(abi.encode(
            _target,
            _data,
            block.timestamp,
            msg.sender
        ));
        
        pendingActions[actionId] = Action({
            target: _target,
            data: _data,
            value: 0,
            description: _description,
            confirmations: 0,
            executeAfter: block.timestamp + ACTION_DELAY,
            executed: false
        });
        
        actionIds.push(actionId);
        emit ActionSubmitted(actionId, _target, _description);
        
        return actionId;
    }

    function confirmAction(bytes32 _actionId) external {
        require(isSigner[msg.sender], "Only signers can confirm");
        require(pendingActions[_actionId].executeAfter > 0, "Action does not exist");
        require(!pendingActions[_actionId].executed, "Already executed");
        require(!confirmations[_actionId][msg.sender], "Already confirmed");
        
        confirmations[_actionId][msg.sender] = true;
        pendingActions[_actionId].confirmations++;
        
        emit ActionConfirmed(_actionId, msg.sender);
    }

    function executeAction(bytes32 _actionId) external {
        Action storage action = pendingActions[_actionId];
        require(action.executeAfter > 0, "Action does not exist");
        require(!action.executed, "Already executed");
        require(block.timestamp >= action.executeAfter, "Delay not met");
        require(action.confirmations >= REQUIRED_CONFIRMATIONS, "Not enough confirmations");
        
        action.executed = true;
        
        (bool success, ) = action.target.call(action.data);
        require(success, "Execution failed");
        
        emit ActionExecuted(_actionId);
    }

    function cancelAction(bytes32 _actionId) external {
        require(isSigner[msg.sender], "Only signers can cancel");
        require(!pendingActions[_actionId].executed, "Already executed");
        
        pendingActions[_actionId].executed = true;
        
        emit ActionCancelled(_actionId);
    }

    function isAuthorized(address _caller, bytes32 _actionId) external view returns (bool) {
        Action storage action = pendingActions[_actionId];
        return isSigner[_caller] 
            && action.confirmations >= REQUIRED_CONFIRMATIONS 
            && !action.executed;
    }

    function getActionIds() external view returns (bytes32[] memory) {
        return actionIds;
    }

    function getAction(bytes32 _actionId) external view returns (
        address target,
        string memory description,
        uint256 confirmations,
        uint256 executeAfter,
        bool executed
    ) {
        Action storage action = pendingActions[_actionId];
        return (
            action.target,
            action.description,
            action.confirmations,
            action.executeAfter,
            action.executed
        );
    }
}
```

| T-24-05 | Spoofing | Governance action submission | Mitigate | Only signers can submit actions |
| T-24-06 | Tampering | Action modification after confirmation | Mitigate | Action data stored immutably, cannot be modified |
| T-24-07 | Repudiation | Signer denies action | Mitigate | All actions emit events, confirmation tracked on-chain |
| T-24-08 | Information Disclosure | Action data visibility | Accept | All action data is public (expected for transparency) |
| T-24-09 | Denial of Service | Governance unresponsive | Mitigate | 24-hour delay allows emergency response, signer rotation possible |
